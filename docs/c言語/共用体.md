## 共用体とは
1. 同じメモリ領域を複数のメンバが共有する構造
2. サイズは最も大きなメンバの大きさになる

## 使い方
~~~c
#include <stdio.h>

union kyouyoutai {
	int16_t sono1;
	char sono2[20];
	double sono3;
} data;

int main() {
	// 初期化
	union kyouyoutai kyoyo = {500};
	printf("%d\n", kyoyo.sono1); // 500
	sprintf(kyoyo.sono2, "kyouyoutai");
	printf("%s\n", kyoyo.sono2); // kyouyoutai
	kyoyo.sono3 = 1.2;
	printf("%lf\n", kyoyo.sono3);
}
~~~
